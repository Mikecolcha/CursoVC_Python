import cv2 as cv
clicked = False
# Funcin encargada de capturar el click del mouse
def onMouse(evento,x,y,flags,param):
    # variable global
    global clicked
    if(evento ==cv.EVENT_LBUTTONUP):
        clicked =True
# Crear objeto cv.VideoCapture()
camCap = cv.VideoCapture(0)
cv.namedWindow("Cmara")
# Configurar que se envie a la funcion onMouse los eventos del mouse
cv.setMouseCallback("Cmara",onMouse)
print("Mostrando Cmara, Presione en la Ventana para detener")
ret,frame = camCap.read()
while(ret and cv.waitKey(1)==-1 and not clicked):
    cv.imshow("Cmara", frame)
    ret,frame = camCap.read()
camCap.release()
cv.destroyAllWindows()
